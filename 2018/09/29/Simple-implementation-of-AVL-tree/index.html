AVL树的简单实现

2018-09-29
chankin
DataStructure
Tree

AVL树是一种自平衡的二叉搜索树（BST），两者之间的区别在于AVL树的左右子树高度之间的差异(平和因子)不能超过1。(树的高度约定为：空结点（空子树）的高度为-1，叶子结点的高度为0)高度和深度有区别,详细了解请看下一下链接

AVL的详细定义:java数据结构与算法之平衡二叉树(AVL树)的设计与实现(其实现和以下代码有点区别,但本质是一样的通过旋转来实现平衡)

实现代码:

public class AVLTree<T extends Comparable<T>>{
    class AVLNode<T extends Comparable<T>> {
        public AVLNode<T> left;
        public AVLNode<T> right;
        public T data;
        public int height;

        public AVLNode(AVLNode<T> left, AVLNode<T> right, T data){
            this.left = left;
            this.right = right;
            this.data = data;
            this.height = 0;
        }
    }
    public  AVLNode<T> root;
    public AVLTree() {
        root = null;
    }
    /*
     * @param x 失衡点
     * @ return w 根节点
     * */
    private AVLNode<T> leftRotation(AVLNode<T> x){
        AVLNode<T> w = x.left;
        x.left = w.right;
        w.right = x;
        w.height = Math.max(height(w.left), height(w.right))+ 1;
        x.height = Math.max(height(x.left), x.height)+ 1;
        return w;
    }

    private AVLNode<T> rightRotation(AVLNode<T> w){
        AVLNode<T> x = w.right;
        w.right = x.left;
        x.left = w;
        w.height = Math.max(height(w.left), height(w.right))+ 1;
        x.height = Math.max(height(x.left), height(x.right))+ 1;
        return x;
    }

    private AVLNode<T> leftRightRotation(AVLNode<T> x){
        x.left = rightRotation(x.left);
        return  leftRotation(x);
    }

    private AVLNode<T> rightLeftRotation(AVLNode<T> x){
        x.right = leftRotation(x.right);
        return rightRotation(x);
    }

    public void insert(T data){
        if(root == null) {
            root = new AVLNode<T>(null,null,data);
            if (data == null) {
                throw new RuntimeException("data can\'t be null");
            }
        }
        root = insert(root,data);//返回根节点
    }

    private AVLNode<T> insert(AVLNode<T> node, T data){// node 为根节点
        if (node == null) {
            // 新建节点
            node = new AVLNode<T>(null,null,data);
            if (node == null) {
                System.out.println("ERROR: create avltree node failed!");
                return null;
            }
        }
        if(data.compareTo(node.data) < 0){
            node.left =  insert(node.left, data);
            if(height(node.left) - height(node.right) == 2){
                if(data.compareTo(node.left.data) < 0){
                    node = leftRotation(node);
                }else{
                    node = leftRightRotation(node);
                }
            }
        }else if( data.compareTo(node.data) > 0 ){
            node.right = insert(node.right, data);
            if(height(node.right) - height(node.left) == 2){
                if(data.compareTo(node.right.data) > 0){
                    node = rightRotation(node);
                }else{
                    node = rightLeftRotation(node);
                }
            }
        } else {
            node.height =Math.max(height(node.left), height(node.right)) + 1;

        }

        return node;
    }
    private int height(AVLNode<T> tree) {
        if (tree != null)
            return tree.height;

        return 0;
    }
    /*
     * 删除操作
     * @param  data
     * @param  node
     * @param return
     */
    public AVLNode<T> remove(T data){
        if(data == null){
            throw new RuntimeException("data can't be null");
        }
        return root = remove(root, data);
    }


    private AVLNode<T> remove(AVLNode<T> node, T data){
        if( node == null) return null;
        if( data.compareTo(node.data) > 0) {
            return remove(node.right, data);
        }else if(data.compareTo(node.data) < 0){
            return remove(node.left, data);
        }


        if (node.left == null || node.right == null) {
            if ((node.left == null) && (node.right == null)) {
                node = null;
            } else if ((node.left != null) && (node.right == null)) {
                node = node.left;
                if(node.height - node.height == 2) {
                    if (data.compareTo(node.left.data) < 0) {
                        node = leftRotation(node);
                    } else {
                        node = leftRightRotation(node);
                    }
                }
            } else if ((node.left == null) && (node.right != null)) {
                node = node.right;
                if(node.right.height - node.left.height == 2){
                    if(data.compareTo(node.data) < 0){
                        node = rightLeftRotation(node);
                    }else{
                        node = rightRotation(node);
                    }
                }
            }
        } else {
            if (node.left.height >node.right.height) {
                node.data = findMax();
                node.left = remove(node.left, node.data);
            } else {
                node.data = findMin();
                node.right = remove(node.right, node.data);
            }
        }
        node.height = Math.max(node.left.height, node.right.height);
        return node;
    }


    public T findMin() {
        AVLNode<T> node = findMin(root);
        if(node != null)
            return node.data;
        return null;
    }
    public T findMax(){
        AVLNode<T> node = findMax(root);
        if(node != null)
            return node.data;
        return null;
    }
    /*
     * 查找最小值结点
     * @param node
     * @return
     */
    private AVLNode<T> findMin(AVLNode<T> node){
        if (node == null)//结束条件
            return null;
        while (node.left !=null)//如果没有左结点,那么t就是最小的
            node = node.left;
        return node;
    }

    /*
     * 查找最小值结点
     * @param node
     * @return
     */
    private AVLNode<T> findMax(AVLNode<T> node){
        if (node == null)//结束条件
            return null;
        while (node.right !=null)//如果没有左结点,那么t就是最小的
            node = node.right;
        return node;
    }
}

        for (int i = 1; i <18 ; i++) {
            avlTree.insert(i);
            System.out.println(i);
        }

    }
